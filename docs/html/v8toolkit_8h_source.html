658 * Returns true on success with the result in the "result" parameter
659 */
660 template<class TupleType = std::tuple<>>
661 v8::Local<v8::Value> call_javascript_function(const v8::Local<v8::Context> context,
662  const v8::Local<v8::Function> function,
663  const v8::Local<v8::Object> receiver,
664  const TupleType & tuple = {})
665 {
666  constexpr int tuple_size = std::tuple_size<TupleType>::value;
667  std::array<v8::Local<v8::Value>, tuple_size> parameters;
668  auto isolate = context->GetIsolate();
669  TupleForEach<tuple_size, TupleType>()(isolate, parameters.data(), tuple);
670 
671  v8::TryCatch tc(isolate);
672 
673  // printf("\n\n**** Call_javascript_function with receiver: %s\n", stringify_value(isolate, v8::Local<v8::Value>::Cast(receiver)).c_str());
674  auto maybe_result = function->Call(context, receiver, tuple_size, parameters.data());
675  if(tc.HasCaught() || maybe_result.IsEmpty()) {
676  printf("Error running javascript function: '%s'\n", *v8::String::Utf8Value(tc.Exception()));
677  throw V8ExecutionException(isolate, tc);
678  }
679  return maybe_result.ToLocalChecked();
680 }

682 /**
683 * Returns true on success with the result in the "result" parameter
684 */
685 template<class TupleType = std::tuple<>>
686 v8::Local<v8::Value> call_javascript_function(const v8::Local<v8::Context> context,
687  const std::string & function_name,
688  const v8::Local<v8::Object> receiver,
689  const TupleType & tuple = {})
690 {
691  auto maybe_value = receiver->Get(context, v8::String::NewFromUtf8(context->GetIsolate(),function_name.c_str()));
692  if(maybe_value.IsEmpty()) {
693  throw InvalidCallException(fmt::format("Function name {} could not be found", function_name));
694  }
695 
696  auto value = maybe_value.ToLocalChecked();
697  if(!value->IsFunction()) {
698  throw InvalidCallException(fmt::format("{} was found but is not a function", function_name));;
699  }
700 
701  return call_javascript_function(context, v8::Local<v8::Function>::Cast(value), receiver, tuple);
702 }
