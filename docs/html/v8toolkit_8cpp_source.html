188 bool get_file_contents(std::string filename, std::string & file_contents, time_t & file_modification_time)
189 {
190  int fd = open(filename.c_str(), O_RDONLY);
191  if (fd == -1) {
192  return false;
193  }
194  struct stat file_stat;
195  if (fstat(fd, &file_stat) == -1) {
196  return false;
197  }
198 
199  file_modification_time = file_stat.st_mtime;
200  auto file_size = file_stat.st_size;
201 
202  file_contents.resize(file_size);
203 
204  auto bytes_remaining = file_size;
205  // TODO: Need to handle file not having as many bytes as expected with open returning 0 before bytes_remaining == 0
206  while (bytes_remaining > 0) {
207  auto bytes_read = read(fd, &file_contents[file_size - bytes_remaining], bytes_remaining);
208  bytes_remaining -= bytes_read;
209  }
210 
211  return true;
212

282 /** Attempts to load the specified external resource.
283 * Attempts to load exact match filename in each path, then <filename>.js in each path, then <filename>.json in each path
284 *
285 * Paths are attempted in vector order[0, 1, 2...]. If the matching filename ends in .js (either because it was specified as such or because
286 * it matched after the suffix was added) it will be executed as a traditional module with the exports object being returned. If the matching
287 * filename ends in .json, the last value in the file will be returned.
288 * The results of require'd files is cached and if the same module (based on the full matching value including path and any added suffix) is
289 * required again, the cached value will be returned. The module will not be re-run.
290 *
291 * The goal of this function is to be as close to node.js require as possible, so patches or descriptions of how it differs are appreciated.
292 * Not that much time was spent trying to determine the exact behavior, so there are likely significant differences
293 */
294 bool require(
295  v8::Local<v8::Context> context,
296  std::string filename,
297  v8::Local<v8::Value> & result,
298  const std::vector<std::string> & paths,
299  bool track_file_modification_times, bool use_cache)
300 {
301  auto isolate = context->GetIsolate();
302  v8::Locker l(isolate);
303  if (filename.find("..") != std::string::npos) {
304  if (V8_TOOLKIT_DEBUG) printf("require() attempted to use a path with more than one . in a row '%s' (disallowed as simple algorithm to stop tricky paths)", filename.c_str());
305  isolate->ThrowException(v8::String::NewFromUtf8(isolate, "Cannot specify a file containing .."));
306  return false;
307  }
308 
309  for (auto suffix : std::vector<std::string>{"", ".js", ".json", }) {
310  for (auto path : paths) {
311  try {
312  auto complete_filename = path + "/" + filename + suffix;
313 
314  std::string file_contents;
315  time_t file_modification_time = 0;
316  if (!get_file_contents(complete_filename, file_contents, file_modification_time)) {
317  if (V8_TOOLKIT_DEBUG) printf("Module not found at %s\n", complete_filename.c_str());
318  continue;
319  }
320 
321  // get the map of cached results for this isolate (guaranteed to exist because it was created before this lambda)
322  // This is read-only and per-isolate and the only way to add is to be in the isolate
323  if (use_cache) {
324  std::lock_guard<std::mutex> l(require_results_mutex);
325  auto & isolate_require_results = require_results[isolate];
326 
327  auto cached_require_results = isolate_require_results.find(complete_filename);
328  if (cached_require_results != isolate_require_results.end()) {
329  if (V8_TOOLKIT_DEBUG) printf("Found cached results, using cache instead of re-running module\n");
330 
331  // if we don't care about file modifications or the file modification time is the same as before,
332  // return the cached result
333  if (!track_file_modification_times || file_modification_time == cached_require_results->second.first) {
334  if (V8_TOOLKIT_DEBUG) printf("Returning cached results\n");
335  result = cached_require_results->second.second.Get(isolate);
336  return true;
337  } else {
338  if (V8_TOOLKIT_DEBUG) printf("Not returning cached results because modification time was no good\n");
339  }
340  } else {
341  if (V8_TOOLKIT_DEBUG) printf("Didn't find cached version for isolate %p %s\n", isolate, complete_filename.c_str());
342  }
343  }
344 
345  // Compile the source code.
346  v8::MaybeLocal<v8::Value> maybe_result;
347 
348  if (std::regex_search(filename, std::regex(".json$"))) {
349  v8::Local<v8::Script> script;
350  v8::Local<v8::Value> error;
351  v8::TryCatch try_catch(isolate);
352  // TODO: make sure requiring a json file is being tested
353  if (V8_TOOLKIT_DEBUG) printf("About to try to parse json: %s\n", file_contents.c_str());
354  maybe_result = v8::JSON::Parse(isolate, v8::String::NewFromUtf8(isolate,file_contents.c_str()));
355  if (try_catch.HasCaught()) {
356  try_catch.ReThrow();
357  if (V8_TOOLKIT_DEBUG) printf("Couldn't run json for %s, error: %s\n", complete_filename.c_str(), *v8::String::Utf8Value(try_catch.Exception()));
358  return false;
359  }
360  } else {
361  v8::Local<v8::Script> script;
362  v8::Local<v8::Value> error;
363  if (!compile_source(context, file_contents, script, error)) {
364  isolate->ThrowException(error);
365  if (V8_TOOLKIT_DEBUG) printf("Couldn't compile .js for %s\n", complete_filename.c_str());
366  return false;
367  }
368 
369  v8::TryCatch try_catch(isolate);
370 
371 
372 
373  // set up the module and exports stuff
374  auto module_object = v8::Object::New(isolate);
375  // TODO: module object should also have "children" array, "filename" string, "id" string, "loaded" boolean, "parent" module object that first loaded this one
376  auto exports_object = v8::Object::New(isolate);
377  add_variable(context, module_object, "exports", exports_object);
378  add_variable(context, context->Global(), "module", module_object);
379  add_variable(context, context->Global(), "exports", exports_object);
380  (void)context->Global()->Set(context, v8::String::NewFromUtf8(isolate, "global"), context->Global());
381 
382 
383  // return value of run doesn't matter, only what is hooked up to export variable
384  (void)script->Run(context);
385 
386  maybe_result = context->Global()->Get(context, v8::String::NewFromUtf8(isolate, "exports"));
387  context->Global()->Delete(context, v8::String::NewFromUtf8(isolate, "module"));
388  context->Global()->Delete(context, v8::String::NewFromUtf8(isolate, "exports"));
389 
390  if (try_catch.HasCaught()) {
391  try_catch.ReThrow();
392  if (V8_TOOLKIT_DEBUG) printf("Couldn't run .js for %s\n", complete_filename.c_str());
393  return false;
394  }
395 
396  }
397 
398  result = maybe_result.ToLocalChecked();
399 
400  // cache the result for subsequent requires of the same module in the same isolate
401  if (use_cache) {
402  std::lock_guard<std::mutex> l(require_results_mutex);
403  auto new_global = new v8::Global<v8::Value>(isolate, result);
404  auto & isolate_require_results = require_results[isolate];
405  isolate_require_results.insert(std::pair<std::string, cached_module_t>(complete_filename, cached_module_t(file_modification_time, *new_global)));
406  }
407  // printf("Require final result: %s\n", stringify_value(isolate, result).c_str());
408  // printf("Require returning resulting object for module %s\n", complete_filename.c_str());
409  return true;
410  }catch(...) {}
411  // if any failures, try the next path if it exists
412  }
413  }
414  // printf("Couldn't find any matches for %s\n", filename.c_str());
415  isolate->ThrowException(v8::String::NewFromUtf8(isolate, "No such module found in any search path"));
416  return false;
417

652 std::string stringify_value(v8::Isolate * isolate, const v8::Local<v8::Value> & value, bool top_level)
653 {
654  static std::vector<v8::Local<v8::Value>> processed_values;
655 
656  if (top_level) {
657  processed_values.clear();
658  };
659 
660  auto context = isolate->GetCurrentContext();
661 
662  std::string output = "";
663 
664  // TODO: This is "slow" but I'm not sure a faster way to work with the types involved
665  for(auto processed_value : processed_values) {
666  if(processed_value == value) {
667  printf("stringify_value already processed '%s', skipping because of cycle\n", *v8::String::Utf8Value(value));
668  return "";
669  }
670  }
671  // printf("Processing %p as unseen\n", *value);
672 
673 
674  processed_values.push_back(value);
675 
676  if(value.IsEmpty()) {
677  // printf("Value IsEmpty\n");
678  return "Value specified as an empty v8::Local";
679  }
680 
681  // if the left is a bool, return true if right is a bool and they match, otherwise false
682  if (value->IsBoolean() || value->IsNumber() || value->IsString() || value->IsFunction() || value->IsUndefined() || value->IsNull()) {
683  // printf("Stringify: treating value as 'normal'\n");
684  output += *v8::String::Utf8Value(value);
685  } else if (value->IsArray()) {
686  // printf("Stringify: treating value as array\n");
687  auto array = v8::Local<v8::Array>::Cast(value);
688  auto array_length = get_array_length(isolate, array);
689 
690  output += "[";
691  auto first_element = true;
692  for (int i = 0; i < array_length; i++) {
693  if (!first_element) {
694  output += ", ";
695  }
696  first_element = false;
697  auto value = array->Get(context, i);
698  output += stringify_value(isolate, value.ToLocalChecked(), false);
699  }
700  output += "]";
701  } else {
702  // printf("Stringify: treating value as object\n");
703  // check this last in case it's some other type of more specialized object we will test the specialization instead (like an array)
704  // objects must have all the same keys and each key must have the same as determined by calling this function on each value
705  // printf("About to see if we can stringify this as an object\n");
706  // print_v8_value_details(value);
707  auto object = v8::Local<v8::Object>::Cast(value);
708  if(value->IsObject() && !object.IsEmpty()) {
709  // printf("Stringify: in object\n");
710  // printf("Stringifying object\n");
711  output += "{";
712  auto keys = make_set_from_object_keys(isolate, object);
713  auto first_key = true;
714  for(auto key : keys) {
715  // printf("Stringify: object key %s\n", key.c_str());
716  if (!first_key) {
717  output += ", ";
718  }
719  first_key = false;
720  output += key;
721  output += ": ";
722  auto value = object->Get(context, v8::String::NewFromUtf8(isolate, key.c_str()));
723  output += stringify_value(isolate, value.ToLocalChecked(), false);
724  }
725  output += "}";
726  }
727  }
728  return output;
729 }
