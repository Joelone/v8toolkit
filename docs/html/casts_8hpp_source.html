705 /**
706 * supports maps containing any type also supported by CastToJS to javascript arrays
707 * It creates an object of key => [values...]
708 * All values are arrays, even if there is only one value in the array.
709 */
710 template<class A, class B, class... Rest> v8::Local<v8::Value>
711 CastToJS<std::multimap<A, B, Rest...>>::operator()(v8::Isolate * isolate, std::multimap<A, B, Rest...> & map){
712  assert(isolate->InContext());
713  auto context = isolate->GetCurrentContext();
714  auto object = v8::Object::New(isolate);
715  for(auto & pair : map){
716  auto key = CastToJS<A&>()(isolate, const_cast<A&>(pair.first));
717  auto value = CastToJS<B&>()(isolate, const_cast<B&>(pair.second));
718 
719  // check to see if a value with this key has already been added
720  bool default_value = true;
721  bool object_has_key = object->Has(context, key).FromMaybe(default_value);
722  if(!object_has_key) {
723  // get the existing array, add this value to the end
724  auto array = v8::Array::New(isolate);
725  (void)array->Set(context, 0, value);
726  (void)object->Set(context, key, array);
727  } else {
728  // create an array, add the current value to it, then add it to the object
729  auto existing_array_value = object->Get(context, key).ToLocalChecked();
730  v8::Handle<v8::Array> existing_array = v8::Handle<v8::Array>::Cast(existing_array_value);
731 
732  //find next array position to insert into (is there no better way to push onto the end of an array?)
733  int i = 0;
734  while(existing_array->Has(context, i).FromMaybe(default_value)){i++;}
735  (void)existing_array->Set(context, i, value);
736  }
737  }
738  return object;
739 }
