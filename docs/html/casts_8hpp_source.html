439 * It creates an object of key => [values...]
440 * All values are arrays, even if there is only one value in the array.
441 */
442 template<class A, class B, class... Rest>
443 struct CastToJS<std::multimap<A, B, Rest...>> {
444  v8::Local<v8::Value> operator()(v8::Isolate * isolate, std::multimap<A, B, Rest...> map){
445  assert(isolate->InContext());
446  auto context = isolate->GetCurrentContext();
447  auto object = v8::Object::New(isolate);
448  for(auto pair : map){
449  auto key = CastToJS<A>()(isolate, pair.first);
450  // v8::Local<v8::String> key = v8::String::NewFromUtf8(isolate, "TEST");
451  auto value = CastToJS<B>()(isolate, pair.second);
452 
453  // check to see if a value with this key has already been added
454  bool default_value = true;
455  bool object_has_key = object->Has(context, key).FromMaybe(default_value);
456  if(!object_has_key) {
457  // get the existing array, add this value to the end
458  auto array = v8::Array::New(isolate);
459  (void)array->Set(context, 0, value);
460  (void)object->Set(context, key, array);
461  } else {
462  // create an array, add the current value to it, then add it to the object
463  auto existing_array_value = object->Get(context, key).ToLocalChecked();
464  v8::Handle<v8::Array> existing_array = v8::Handle<v8::Array>::Cast(existing_array_value);
465 
466  //find next array position to insert into (is there no better way to push onto the end of an array?)
467  int i = 0;
468  while(existing_array->Has(context, i).FromMaybe(default_value)){i++;}
469  (void)existing_array->Set(context, i, value);
470  }
471  }
472  return object;
473  }
