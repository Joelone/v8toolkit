388 template<class A, class B, class... Rest>
389 struct CastToJS<std::multimap<A, B, Rest...>> {
390  v8::Local<v8::Value> operator()(v8::Isolate * isolate, std::multimap<A, B, Rest...> map){
391  assert(isolate->InContext());
392  auto context = isolate->GetCurrentContext();
393  auto object = v8::Object::New(isolate);
394  for(auto pair : map){
395  auto key = CastToJS<A>()(isolate, pair.first);
396  // v8::Local<v8::String> key = v8::String::NewFromUtf8(isolate, "TEST");
397  auto value = CastToJS<B>()(isolate, pair.second);
398 
399  // check to see if a value with this key has already been added
400  bool default_value = true;
401  bool object_has_key = object->Has(context, key).FromMaybe(default_value);
402  if(!object_has_key) {
403  // get the existing array, add this value to the end
404  auto array = v8::Array::New(isolate);
405  (void)array->Set(context, 0, value);
406  (void)object->Set(context, key, array);
407  } else {
408  // create an array, add the current value to it, then add it to the object
409  auto existing_array_value = object->Get(context, key).ToLocalChecked();
410  v8::Handle<v8::Array> existing_array = v8::Handle<v8::Array>::Cast(existing_array_value);
411 
412  //find next array position to insert into (is there no better way to push onto the end of an array?)
413  int i = 0;
414  while(existing_array->Has(context, i).FromMaybe(default_value)){i++;}
415  (void)existing_array->Set(context, i, value);
416  }
417  }
418  return object;
419  }
